## 1. How to Handle Errors
Error handling ensures your program doesn't crash unexpectedly.  
- Use `try...catch` for synchronous code  
- Use `.catch()` for promises  
- Use `finally` for cleanup  

**Example:**  
```javascript
// Synchronous error handling
try {
  let result = riskyFunction();
  console.log(result);
} catch (error) {
  console.error('An error occurred:', error.message);
} finally {
  console.log('Cleanup code always runs');
}

// Asynchronous error handling
fetchData()
  .then(data => console.log(data))
  .catch(error => console.error('Fetch failed:', error));
```

## 2. What is Promisification
Promisification is converting a callback-based function into a Promise-based one.  
Example: Using `util.promisify` in Node.js  

**Example:**  
```javascript
const fs = require('fs');
const util = require('util');

// Original callback-based function
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});

// Promisified version
const readFilePromise = util.promisify(fs.readFile);

readFilePromise('file.txt', 'utf8')
  .then(data => console.log(data))
  .catch(err => console.error(err));
```

## 3. Map, Reduce, and Filter Functions
- `map()` → transforms each element  
- `filter()` → selects elements based on condition  
- `reduce()` → reduces array to a single value  

**Examples:**  
```javascript
const numbers = [1, 2, 3, 4, 5];

// map: double each number
const doubled = numbers.map(num => num * 2); // [2, 4, 6, 8, 10]

// filter: get even numbers
const evens = numbers.filter(num => num % 2 === 0); // [2, 4]

// reduce: sum all numbers
const sum = numbers.reduce((acc, num) => acc + num, 0); // 15
```

## 4. Types of Functions
- Function Declaration  
- Function Expression  
- Arrow Function  
- Anonymous Function  
- Higher-order Function  

**Examples:**  
```javascript
// Function Declaration
function greet(name) {
  return `Hello, ${name}!`;
}

// Function Expression
const greetExpr = function(name) {
  return `Hello, ${name}!`;
};

// Arrow Function
const greetArrow = (name) => `Hello, ${name}!`;

// Anonymous Function (used as callback)
setTimeout(function() {
  console.log('Anonymous function');
}, 1000);

// Higher-order Function (takes function as argument)
function applyOperation(num, operation) {
  return operation(num);
}
const result = applyOperation(5, (x) => x * 2); // 10
```

## 5. What is Currying
Currying is transforming a function with multiple arguments into nested functions.  

**Example:**  
```javascript
// Regular function
function add(a, b, c) {
  return a + b + c;
}

// Curried version
function curriedAdd(a) {
  return function(b) {
    return function(c) {
      return a + b + c;
    };
  };
}

// Usage
const result = curriedAdd(1)(2)(3); // 6

// Or with arrow functions
const curriedAddArrow = a => b => c => a + b + c;
const result2 = curriedAddArrow(1)(2)(3); // 6
```

## 6. Promise Chaining Method
Using `.then()` repeatedly to execute async operations in sequence.  

**Example:**  
```javascript
function fetchUser(id) {
  return fetch(`/api/user/${id}`).then(res => res.json());
}

function fetchPosts(userId) {
  return fetch(`/api/posts?userId=${userId}`).then(res => res.json());
}

function processData(data) {
  console.log('Processed:', data);
  return data;
}

// Chaining promises
fetchUser(1)
  .then(user => {
    console.log('User:', user);
    return fetchPosts(user.id);
  })
  .then(posts => {
    console.log('Posts:', posts);
    return processData(posts);
  })
  .then(result => console.log('Final result:', result))
  .catch(error => console.error('Error:', error));
```

## 7. What is DOM


## 8. Object.seal() Method
Prevents adding/removing properties but allows modifying existing ones.  

**Example:**  
```javascript
const obj = { name: 'John', age: 30 };

console.log(Object.isSealed(obj)); // false

Object.seal(obj);

obj.age = 31; // Allowed: modifying existing property
console.log(obj.age); // 31

obj.city = 'New York'; // Not allowed: adding new property
console.log(obj.city); // undefined

delete obj.name; // Not allowed: deleting property
console.log(obj.name); // 'John'

console.log(Object.isSealed(obj)); // true
```

## 9. What is Callback Hell
Nested callbacks that make code hard to read and maintain.  

**Example (Callback Hell):**  
```javascript
// Nested callbacks - hard to read
getUser(1, function(user) {
  getPosts(user.id, function(posts) {
    getComments(posts[0].id, function(comments) {
      getLikes(comments[0].id, function(likes) {
        console.log('Likes:', likes);
      }, function(error) { console.error(error); });
    }, function(error) { console.error(error); });
  }, function(error) { console.error(error); });
}, function(error) { console.error(error); });

// Better with Promises
getUser(1)
  .then(user => getPosts(user.id))
  .then(posts => getComments(posts[0].id))
  .then(comments => getLikes(comments[0].id))
  .then(likes => console.log('Likes:', likes))
  .catch(error => console.error(error));
```

## 10. NaN + 1 in JavaScript
Result is still `NaN` because any operation with `NaN` returns `NaN`.  

**Example:**  
```javascript
console.log(NaN + 1); // NaN
console.log(NaN * 2); // NaN
console.log(NaN / 0); // NaN

// Checking for NaN
console.log(isNaN(NaN + 1)); // true
console.log(Number.isNaN(NaN + 1)); // true

// NaN is not equal to itself
console.log(NaN === NaN); // false
```

## 11. Prototypal Inheritance
Objects inherit properties from other objects using prototypes.  

**Example:**  
```javascript
// Constructor function
function Animal(name) {
  this.name = name;
}

Animal.prototype.speak = function() {
  console.log(`${this.name} makes a sound`);
};

// Child constructor
function Dog(name, breed) {
  Animal.call(this, name); // Call parent constructor
  this.breed = breed;
}

// Inherit prototype
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;

// Add method to child
Dog.prototype.bark = function() {
  console.log(`${this.name} barks`);
};

const dog = new Dog('Buddy', 'Golden Retriever');
dog.speak(); // "Buddy makes a sound" (inherited)
dog.bark();  // "Buddy barks" (own method)
```

## 12. CASE Syntax in SQL
Used for conditional logic in queries.  

**Example:**  
```sql
-- Simple CASE
SELECT name,
       CASE department
         WHEN 'IT' THEN 'Information Technology'
         WHEN 'HR' THEN 'Human Resources'
         WHEN 'FIN' THEN 'Finance'
         ELSE 'Other'
       END AS department_name
FROM employees;

-- Searched CASE
SELECT name, salary,
       CASE
         WHEN salary < 30000 THEN 'Low'
         WHEN salary BETWEEN 30000 AND 60000 THEN 'Medium'
         WHEN salary > 60000 THEN 'High'
         ELSE 'Unknown'
       END AS salary_range
FROM employees;

-- Using CASE in aggregate functions
SELECT 
  SUM(CASE WHEN status = 'active' THEN 1 ELSE 0 END) AS active_count,
  SUM(CASE WHEN status = 'inactive' THEN 1 ELSE 0 END) AS inactive_count
FROM users;
```

## 13. Call Stack and Heap
- Call Stack → handles function execution  
- Heap → stores objects and variables  

**Example:**  
```javascript
function multiply(a, b) {
  return a * b; // Primitive value stored in stack
}

function calculate() {
  const obj = { value: 10 }; // Object stored in heap
  const result = multiply(5, 3); // Function call added to stack
  return result;
}

console.log(calculate()); // 15

// Call Stack (LIFO):
// 1. main() - global execution context
// 2. calculate() - function execution context
// 3. multiply(5, 3) - function execution context
// 4. multiply() returns, popped from stack
// 5. calculate() returns, popped from stack

// Heap: stores the object { value: 10 }
```

## 14. What is JSON
JavaScript Object Notation, used for data exchange.  

**Example:**  
```javascript
// JSON string
const jsonString = '{"name": "John", "age": 30, "city": "New York"}';

// Parsing JSON to JavaScript object
const person = JSON.parse(jsonString);
console.log(person.name); // "John"

// Converting JavaScript object to JSON
const car = {
  make: "Toyota",
  model: "Camry",
  year: 2020,
  features: ["navigation", "bluetooth"]
};

const carJson = JSON.stringify(car);
console.log(carJson);
// {"make":"Toyota","model":"Camry","year":2020,"features":["navigation","bluetooth"]}

// Handling dates (JSON doesn't support dates natively)
const dataWithDate = {
  event: "Conference",
  date: new Date()
};

const jsonWithDate = JSON.stringify(dataWithDate, (key, value) => {
  return value instanceof Date ? value.toISOString() : value;
});
```

## 15. Necessity of Handling Errors
Prevents crashes, improves user experience, and helps debugging.  

**Example:**  
```javascript
// Without error handling
function divide(a, b) {
  return a / b;
}
console.log(divide(10, 0)); // Infinity (unexpected)

// With error handling
function safeDivide(a, b) {
  if (b === 0) {
    throw new Error('Division by zero is not allowed');
  }
  return a / b;
}

try {
  console.log(safeDivide(10, 0));
} catch (error) {
  console.error('Error:', error.message); // "Division by zero is not allowed"
  // Can show user-friendly message or fallback behavior
}

// Async error handling
async function fetchData() {
  try {
    const response = await fetch('/api/data');
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    const data = await response.json();
    return data;
  } catch (error) {
    console.error('Failed to fetch data:', error);
    // Return default data or show error to user
    return { error: 'Unable to load data' };
  }
}
```
